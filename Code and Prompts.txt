Prompts:
Build a complete web-based prototype named "CSP-Guard" that analyzes copied code snippets and determines whether they are 
safe when reused in a new context using AI-assisted semantic analysis and context comparison.

The system should follow this pipeline:

1. Perform clone detection between source and destination code using token similarity, AST similarity, and data flow similarity.
2. Extract execution context from both code snippets.
3. Compute Execution Context Delta (ECD), representing differences between source and destination contexts.
4. Calculate a multi-dimensional risk score.
5. Generate a detailed report with explanation and suggested fixes.

Execution Context Delta (ECD) must analyze:

* Variables (V): type, scope, mutability
* Scope bindings (S)
* Exception handling (E)
* Taint (T): user input or unsafe data flow
* Resource usage (R)

The UI must include:

* Title: CSP-Guard
* Short project description
* Input box for Source Code
* Input box for Destination Code
* "Analyze Risk" button
* Output section displaying:

  * Risk Score (0 to 1)
  * Risk Level (Safe, Low, Medium, High)
  * Explanation of risk
  * Suggested fixes

Risk levels should be:

* 0.00–0.15: Safe
* 0.15–0.40: Low Risk
* 0.40–0.70: Medium Risk
* 0.70–1.00: High Risk

The system must include these architecture components:
* Parser
* Clone Detector
* Context Extractor
* ECD Comparator
* Risk Engine
* Report Generator

Integrate an AI API (use OpenRouter) to:

* Perform semantic analysis of code
* Compare context between snippets
* Generate human-readable explanations and fixes

Include this statement in the UI:
"AI-assisted context-aware risk evaluation using LLM"

The implementation should:

* Be a single-page web application (HTML, CSS, JavaScript)
* Work without a backend (API calls directly from frontend)
* Be clean, minimal, and responsive
* Clearly separate logic for clone detection, ECD computation, and risk scoring

Also include sample test cases (e.g., Python, C, JavaScript snippets with safe and unsafe reuse) and display their scores.
Ensure the output is visually clear and easy to understand, with proper formatting and structured sections.

function clamp(value, min, max) {

function normalize(code) {
  return code
    .replace(/\/\/.*?$|\/\*[\s\S]*?\*\/|#.*?$/gm, " ")
    .replace(/"(?:\\.|[^"\\])*"|'(?:\\.|[^'\\])*'/g, " STRING ");
}

function tokenize(code) {
  return normalize(code).match(/[A-Za-z_][A-Za-z0-9_]*|\d+|==|!=|<=|>=|=>|[{}()[\];,.:+\-*/%<>!=]/g)?.map(token => token.toLowerCase()) || [];
}

function jaccard(listA, listB) {
  const setA = new Set(listA);
  const setB = new Set(listB);
  const union = new Set([...setA, ...setB]);
  let intersection = 0;
  for (const token of setA) if (setB.has(token)) intersection += 1;
  return union.size ? intersection / union.size : 0;
}

function freqMap(list) {
  const map = new Map();
  for (const item of list) map.set(item, (map.get(item) || 0) + 1);
  return map;
}

function cosineSimilarity(listA, listB) {
  const left = freqMap(listA);
  const right = freqMap(listB);
  const keys = new Set([...left.keys(), ...right.keys()]);
  let dot = 0;
  let magA = 0;
  let magB = 0;
  for (const key of keys) {
    const a = left.get(key) || 0;
    const b = right.get(key) || 0;
    dot += a * b;
    magA += a * a;
    magB += b * b;
  }
  return magA && magB ? dot / (Math.sqrt(magA) * Math.sqrt(magB)) : 0;
}

function structuralSignature(code) {
  return tokenize(code).map(token => {
    if (/^\d+$/.test(token)) return "NUM";
    if (["if", "for", "while", "switch", "try", "catch", "function", "def", "class", "return", "const", "let", "var", "int", "String"].includes(token)) return token;
    if (/^[{}()[\];,.:+\-*/%<>!=]+$/.test(token)) return token;
    return "ID";
  });
}

function extractBlocks(code) {
  const lines = code.split(/\r?\n/).map(line => line.trim()).filter(Boolean);
  const blocks = [];
  let current = [];
  let depth = 0;
  for (const line of lines) {
    current.push(line.replace(/\s+/g, " "));
    depth += (line.match(/[({[]/g) || []).length;
    depth -= (line.match(/[)}\]]/g) || []).length;
    if (depth <= 0) {
      blocks.push(current.join(" "));
      current = [];
      depth = 0;
    }
  }
  if (current.length) blocks.push(current.join(" "));
  return blocks.length ? blocks : lines.slice(0, 12);
}

function astSimilarity(source, destination) {
  const blocksA = extractBlocks(source).map(block => block.toLowerCase());
  const blocksB = extractBlocks(destination).map(block => block.toLowerCase());
  const sigA = structuralSignature(source);
  const sigB = structuralSignature(destination);
  return clamp((0.55 * jaccard(blocksA, blocksB) + 0.45 * cosineSimilarity(blocksA, blocksB) + 0.65 * jaccard(sigA, sigB) + 0.35 * cosineSimilarity(sigA, sigB)) / 2, 0, 1);
}

function extractDataFlow(code) {
  const identifiers = code.match(/\b[A-Za-z_][A-Za-z0-9_]*\b/g) || [];
  const usage = new Set(identifiers.map(item => item.toLowerCase()));
  return {
    identifiers: [...usage],
    taint: /request|input|argv|stdin|prompt|body|params|cookies|headers|fetch\(/i.test(code),
    sink: /eval\(|exec\(|innerHTML|document\.write|execute\(|gets\(|system\(|sprintf|strcpy/i.test(code)
  };
}

function inferContext(code) {
  const lines = code.split(/\r?\n/);
  const variables = new Map();
  for (const line of lines) {
    const match = line.match(/\b(?:const|let|var|int|float|double|char|bool|boolean|String|str)\s+([A-Za-z_][A-Za-z0-9_]*)/);
    if (match) {
      variables.set(match[1].toLowerCase(), {
        type: /String|str/.test(line) ? "string" : /int|float|double|number/.test(line) ? "number" : "unknown",
        scope: /function|def/.test(line) ? "function" : "module",
        mutability: /const|final/.test(line) ? "immutable" : "mutable"
      });
    }
  }

  return {
    tokens: tokenize(code),
    variables: [...variables.entries()].map(([name, meta]) => ({ name, ...meta })),
    bindings: [...variables.keys()],
    exceptionHandling: /try\b|catch\b|except\b|finally\b/i.test(code),
    taint: {
      sources: [...code.matchAll(/\b(request|input|argv|stdin|prompt|body|params|cookies|headers)\b/gi)].map(match => match[1].toLowerCase()),
      sinks: [...code.matchAll(/\b(eval|exec|innerHTML|document\.write|execute|gets|system|sprintf|strcpy)\b/gi)].map(match => match[1].toLowerCase())
    },
    resources: {
      file: /open\(|fopen\(|readFile|writeFile|fs\.|File\b/i.test(code),
      network: /fetch\(|axios|requests\.|http\.|socket|XMLHttpRequest/i.test(code),
      process: /spawn|system\(|subprocess|exec\(|os\.system/i.test(code)
    },
    dataFlow: extractDataFlow(code)
  };
}

function compareContexts(sourceContext, destinationContext) {
  const sourceVars = new Map(sourceContext.variables.map(item => [item.name, item]));
  const destVars = new Map(destinationContext.variables.map(item => [item.name, item]));
  const names = new Set([...sourceVars.keys(), ...destVars.keys()]);

  let variableDelta = 0;
  const drift = [];
  for (const name of names) {
    const left = sourceVars.get(name);
    const right = destVars.get(name);
    if (!left || !right) {
      variableDelta += 0.6;
      drift.push(`${name}: added/removed`);
      continue;
    }
    const changes = [];
    if (left.type !== right.type) changes.push(`type ${left.type} -> ${right.type}`);
    if (left.scope !== right.scope) changes.push(`scope ${left.scope} -> ${right.scope}`);
    if (left.mutability !== right.mutability) changes.push(`mutability ${left.mutability} -> ${right.mutability}`);
    if (changes.length) {
      variableDelta += 0.3 * changes.length;
      drift.push(`${name}: ${changes.join(", ")}`);
    }
  }

  const bindings = new Set([...sourceContext.bindings, ...destinationContext.bindings]);
  const overlap = [...bindings].filter(name => sourceContext.bindings.includes(name) && destinationContext.bindings.includes(name)).length;
  const scopeDelta = bindings.size ? 1 - overlap / bindings.size : 0;
  const resourceDelta = Number(sourceContext.resources.file !== destinationContext.resources.file) + Number(sourceContext.resources.network !== destinationContext.resources.network) + Number(sourceContext.resources.process !== destinationContext.resources.process);

  return {
    variableDelta: clamp(variableDelta / Math.max(1, names.size), 0, 1),
    scopeDelta: clamp(scopeDelta, 0, 1),
    exceptionDelta: sourceContext.exceptionHandling === destinationContext.exceptionHandling ? 0 : 1,
    taintDelta: (sourceContext.taint.sources.length !== destinationContext.taint.sources.length || sourceContext.taint.sinks.length !== destinationContext.taint.sinks.length) ? 1 : 0,
    resourceDelta: clamp(resourceDelta / 3, 0, 1),
    drift
  };
}

function riskBand(score) {
  if (score < 0.15) return { label: "Safe", color: "#7ef0a9", background: "rgba(126, 240, 169, 0.12)" };
  if (score < 0.4) return { label: "Low Risk", color: "#15d7ff", background: "rgba(21, 215, 255, 0.12)" };
  if (score < 0.7) return { label: "Medium Risk", color: "#ffbb6d", background: "rgba(255, 187, 109, 0.12)" };
  return { label: "High Risk", color: "#ff6d8d", background: "rgba(255, 109, 141, 0.12)" };
}

function localAnalysis(source, destination) {
  const sourceContext = inferContext(source);
  const destinationContext = inferContext(destination);
  const tokenSimilarity = jaccard(sourceContext.tokens, destinationContext.tokens);
  const structuralSimilarity = astSimilarity(source, destination);
  const dataFlowSimilarity = clamp(jaccard(sourceContext.dataFlow.identifiers, destinationContext.dataFlow.identifiers) * 0.75 + cosineSimilarity(sourceContext.dataFlow.identifiers, destinationContext.dataFlow.identifiers) * 0.25, 0, 1);
  const cloneScore = clamp(tokenSimilarity * 0.4 + structuralSimilarity * 0.35 + dataFlowSimilarity * 0.25, 0, 1);
  const contextDelta = compareContexts(sourceContext, destinationContext);
  const deltaScore = clamp(contextDelta.variableDelta * 0.32 + contextDelta.scopeDelta * 0.2 + contextDelta.exceptionDelta * 0.12 + contextDelta.taintDelta * 0.22 + contextDelta.resourceDelta * 0.14, 0, 1);
  let risk = clamp(cloneScore * 0.15 + deltaScore * 0.65, 0, 1);
  if (destinationContext.dataFlow.taint && destinationContext.dataFlow.sink) risk = clamp(risk + 0.28, 0, 1);
  if (/eval\(|exec\(|gets\(|system\(|innerHTML|document\.write/i.test(destination)) risk = clamp(risk + 0.18, 0, 1);

  return {
    sourceContext,
    destinationContext,
    tokenSimilarity,
    structuralSimilarity,
    dataFlowSimilarity,
    cloneScore,
    contextDelta,
    deltaScore,
    risk
  };
}

function explain(assessment) {
  const notes = [];
  if (assessment.cloneScore > 0.55) notes.push("High structural overlap.");
  if (assessment.contextDelta.variableDelta > 0.3) notes.push("Variable shape changed.");
  if (assessment.contextDelta.scopeDelta > 0.25) notes.push("Scope bindings shifted.");
  if (assessment.contextDelta.exceptionDelta) notes.push("Error handling changed.");
  if (assessment.contextDelta.taintDelta) notes.push("User input reaches a different sink path.");
  if (assessment.contextDelta.resourceDelta > 0.2) notes.push("Resource access changed.");
  return notes.length ? notes.join(" ") : "The snippets align closely and the execution-context delta is small.";
}

function fixes(assessment) {
  const items = [];
  if (assessment.contextDelta.taintDelta) items.push("Sanitize user-controlled input before it reaches a sink.");
  if (assessment.contextDelta.scopeDelta) items.push("Rebind copied variables instead of assuming the original scope.");
  if (assessment.contextDelta.exceptionDelta) items.push("Mirror the source error handling in the destination context.");
  if (assessment.contextDelta.resourceDelta) items.push("Reapply file, network, and process safety checks.");
  if (!items.length) items.push("Keep the logic, but re-run validation and tests in the destination context.");
  return items.slice(0, 3);
}

function escapeHtml(value) {
  return value
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#39;");
}

async function requestAI(source, destination, assessment) {
  if (!state.useAI || !state.apiKey) return null;

  try {
    const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${state.apiKey}`,
        "HTTP-Referer": location.origin === "null" ? "http://localhost" : location.origin,
        "X-Title": "CSP-Guard"
      },
      body: JSON.stringify({
        model: "openrouter/free",
        messages: [
          {
            role: "system",
            content: "Explain code reuse risk in plain language and return JSON with explanation and fixes."
          },
          {
            role: "user",
            content: JSON.stringify({ source, destination, localAssessment: assessment }, null, 2)
          }
        ],
        temperature: 0.2
      })
    });

    if (!response.ok) throw new Error(await response.text() || `AI request failed with status ${response.status}`);
    const data = await response.json();
    const text = data?.choices?.[0]?.message?.content || "";
    if (!text) return null;
    try {
      return JSON.parse(text);
    } catch {
      return { explanation: text.trim(), fixes: [] };
    }
  } catch (error) {
    DOM.statusText.textContent = `AI unavailable, using local analysis: ${error.message}`;
    return null;
  }
}

function renderReport(assessment, aiResult) {
  const band = riskBand(assessment.risk);
  DOM.riskScore.textContent = assessment.risk.toFixed(2);
  DOM.riskBadge.textContent = band.label;
  DOM.riskBadge.style.background = band.background;
  DOM.riskBadge.style.color = band.color;
  DOM.scoreRing.style.stroke = band.color;
  DOM.scoreRing.style.strokeDashoffset = String(RING_CIRCUMFERENCE * (1 - assessment.risk));

  DOM.cloneValue.textContent = `${Math.round(assessment.cloneScore * 100)}%`;
  DOM.deltaValue.textContent = `${Math.round(assessment.deltaScore * 100)}%`;
  DOM.levelValue.textContent = `${band.label} (${Math.round(assessment.risk * 100)}%)`;

  const explanation = aiResult?.explanation || explain(assessment);
  const fixList = Array.isArray(aiResult?.fixes) && aiResult.fixes.length ? aiResult.fixes : fixes(assessment);

  DOM.reportBody.innerHTML = `
    <section class="report-section wide">
      <h3>Risk Overview</h3>
      <p>${escapeHtml(explanation)}</p>
      <div class="signal-row">
        <span>Clone ${Math.round(assessment.cloneScore * 100)}%</span>
        <span>ECD ${Math.round(assessment.deltaScore * 100)}%</span>
        <span>${band.label} ${Math.round(assessment.risk * 100)}%</span>
      </div>
    </section>
    <section class="report-section">
      <h3>Clone + ECD</h3>
      <ul>
        <li>Token: ${Math.round(assessment.tokenSimilarity * 100)}%</li>
        <li>AST: ${Math.round(assessment.structuralSimilarity * 100)}%</li>
        <li>Data flow: ${Math.round(assessment.dataFlowSimilarity * 100)}%</li>
        <li>V/S/E/T/R: ${Math.round(assessment.contextDelta.variableDelta * 100)} / ${Math.round(assessment.contextDelta.scopeDelta * 100)} / ${Math.round(assessment.contextDelta.exceptionDelta * 100)} / ${Math.round(assessment.contextDelta.taintDelta * 100)} / ${Math.round(assessment.contextDelta.resourceDelta * 100)}%</li>
      </ul>
    </section>
    <section class="report-section">
      <h3>Suggested Fixes</h3>
      <ul>${fixList.map(item => `<li>${escapeHtml(item)}</li>`).join("")}</ul>
    </section>
  `;
}

  const sampleButtons = SAMPLE_CASES.map(sample => {
    const assessment = localAnalysis(sample.source, sample.destination);
    return `
      <button class="sample-chip" data-sample-id="${sample.id}" type="button">
        ${sample.label} [${sample.type}] ${assessment.risk.toFixed(2)}
      </button>
    `;
  }).join("");

  DOM.sampleStrip.innerHTML = sampleButtons;

  document.querySelectorAll("[data-sample-id]").forEach(button => {
    button.addEventListener("click", () => {
      const sample = SAMPLE_CASES.find(entry => entry.id === button.dataset.sampleId);
      if (!sample) return;
      DOM.sourceCode.value = sample.source;
      DOM.destinationCode.value = sample.destination;

  });
}

function setInitialSample(sample) {
  DOM.sourceCode.value = sample.source;
  DOM.destinationCode.value = sample.destination;
  DOM.statusText.textContent = `Loaded sample: ${sample.label}`;
}

async function analyze() {
  const source = DOM.sourceCode.value.trim();
  const destination = DOM.destinationCode.value.trim();

  if (!source || !destination) {
    DOM.statusText.textContent = "Paste both source and destination code first.";
    DOM.reportBody.innerHTML = '<div class="empty-state">Both code inputs are required before analysis.</div>';
    return;
  }

  DOM.analyzeBtn.disabled = true;
  DOM.analyzeBtn.textContent = "Analyzing...";
  DOM.statusText.textContent = "Running clone detection, ECD computation, and risk scoring...";

  try {
    const assessment = localAnalysis(source, destination);
    const aiResult = await requestAI(source, destination, assessment);
    renderReport(assessment, aiResult);
    DOM.statusText.textContent = `Analysis complete. Risk level: ${riskBand(assessment.risk).label}.`;
  } catch (error) {
    DOM.statusText.textContent = `Analysis failed: ${error.message}`;
    DOM.reportBody.innerHTML = `<div class="empty-state">${escapeHtml(error.message)}</div>`;
  } finally {
    DOM.analyzeBtn.disabled = false;
    DOM.analyzeBtn.textContent = "Analyze Risk";
  }
}

  const input = document.createElement("input");
  input.type = "file";
  input.accept = ".txt,.key,text/plain";
  input.addEventListener("change", async () => {
    const file = input.files?.[0];
    if (!file) return;
    const text = await file.text();
    const key = text.trim().split(/\r?\n/).find(Boolean) || "";
    DOM.apiKeyInput.value = key;
    DOM.statusText.textContent = "AI key loaded from file.";
    safeStorageSet("cspGuardApiKey", key);
  }, { once: true });
  input.click();
}
